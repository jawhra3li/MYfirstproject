Lexicon setup

The three category sets are defined independently, then explicitly de-overlapped with set subtraction — a defensive step so no single string can be tagged under two categories at once:

PERSON_LIST = {
    'Sarah Johnson', 'Dr. Ahmed Al Balushi', 'Emily Carter',
    'Michael Brown', 'Fatima Al Hinai',
}

ORG_LIST = {
    'TechNova Solutions', 'Future Vision Labs', 'Global Health Institute',
    'Green Energy Corporation', 'Digital Future Academy',
}

LOCATION_LIST = {
    'Muscat', 'Oman', 'Dubai', 'London', 'New York City', 'Salalah',
}

# Ensure no entity is duplicated across categories.
PERSON_LIST = set(PERSON_LIST) - set(ORG_LIST) - set(LOCATION_LIST)
ORG_LIST = set(ORG_LIST) - set(PERSON_LIST) - set(LOCATION_LIST)
LOCATION_LIST = set(LOCATION_LIST) - set(PERSON_LIST) - set(ORG_LIST)

Key logic — full simple_ner()

The complete function: normalize whitespace, sweep the three lexicons in priority order (PERSON → ORGANIZATION → LOCATION) with longest-match-first and span masking, then run the three date/time regex patterns against the original text before de-duplicating:

simple_ner(text)python
def simple_ner(text):
    """Extract PERSON, ORGANIZATION, LOCATION, and DATE/TIME entities from text."""
    results = []
    text_clean = re.sub(r'\s+', ' ', text)

    def find_entities(lexicon, label):
        nonlocal text_clean
        # Longest-first so multi-word entities are claimed before their
        # shorter substrings (e.g. "Dr. Ahmed Al Balushi" before "Ahmed").
        sorted_lexicon = sorted(list(lexicon), key=len, reverse=True)
        for entity in sorted_lexicon:
            if len(entity.strip()) <= 1:
                continue
            pattern = r'\b' + re.escape(entity) + r'\b'
            for m in re.finditer(pattern, text_clean, re.IGNORECASE):
                match = m.group().strip()
                results.append((match, label))
                # Mask the matched span so it can't be re-matched by a
                # shorter, overlapping lexicon entry later in the loop.
                text_clean = text_clean[:m.start()] + (" " * len(match)) + text_clean[m.end():]

    find_entities(PERSON_LIST, "PERSON")
    find_entities(ORG_LIST, "ORGANIZATION")
    find_entities(LOCATION_LIST, "LOCATION")

    date_patterns = [
        r'\d{1,2}\s+(?:January|February|March|April|May|June|July|August|September|October|November|December)\s+\d{4}',
        r'(?:January|February|March|April|May|June|July|August|September|October|November|December)\s+\d{1,2},\s+\d{4}',
        r'\d{1,2}/\d{1,2}/\d{2,4}',
    ]
    for pattern in date_patterns:
        for m in re.finditer(pattern, text, re.IGNORECASE):
            results.append((m.group(), "DATE/TIME"))

    return list(dict.fromkeys(results))  # de-dupe, preserve order

Evaluation function

Aggregates true/false positives and false negatives across all paragraphs via set operations before computing precision, recall, and F1 — not a per-paragraph average, which would weight short and long paragraphs equally rather than by entity count:

evaluate_ner(expected, predicted)python
def evaluate_ner(expected, predicted):
    """Compare predicted vs. expected (entity, label) tuples per paragraph."""
    total_tp = 0
    total_fp = 0
    total_fn = 0

    for exp_par, pred_par in zip(expected, predicted):
        exp_set = set(exp_par)
        pred_set = set(pred_par)

        tp = len(exp_set & pred_set)
        fp = len(pred_set - exp_set)
        fn = len(exp_set - pred_set)

        total_tp += tp
        total_fp += fp
        total_fn += fn

    precision = total_tp / (total_tp + total_fp) if (total_tp + total_fp) else 0
    recall = total_tp / (total_tp + total_fn) if (total_tp + total_fn) else 0
    f1 = 2 * precision * recall / (precision + recall) if (precision + recall) else 0

    print(f"TP: {total_tp} | FP: {total_fp} | FN: {total_fn}")
    print(f"Precision: {precision:.2%}")
    print(f"Recall   : {recall:.2%}")
    print(f"F1 Score : {f1:.2%}")

Main execution block

Guarded behind __name__ == "__main__" so the module can be imported elsewhere (e.g. unit tests) without immediately running the demo:

run blockpython
if __name__ == "__main__":
    predicted = [simple_ner(t) for t in texts]

    evaluate_ner(expected, predicted)

    for i, (text, pred) in enumerate(zip(texts, predicted)):
        print(f"\n--- Paragraph {i + 1} ---")
        print(f"Text Preview: {text[:80]}...")
        if not pred:
            print("  No entities found.")
        else:
            for entity, label in pred:
                print(f"  • {entity} => {label}")
